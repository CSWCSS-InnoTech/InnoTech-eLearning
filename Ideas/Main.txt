    Assert(Coins.Count == 5);
    Pirate.Flung += (Angle Direction) => { Pirate.Direction = Direction; };
    Pirate.BounceOnEdgeTouched = true;
    Device.StartTimer(TimeSpan.FromMilliSeconds(2000),
      () => { foreach(Coin Item in Coins) Item.MoveTo(RandomLocation - Item.Size); });
    foreach(Coin Item in Coins) Item.DisappearIfCollidedWith(Pirate);
    Reset.Clicked += () => { foreach(Coin Item in Coins) Item.Visible = true; };
    HackOn.Clicked += () => { Pirate.Speed = 30; };
    HackOff.Clicked += () => { Pirate.Speed = 10; };
